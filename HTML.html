    <script>
        const scenarios = [
            {
                id: 1,
                community: ['K♠', 'K♥', 'K♣'],
                hand: ['A♠', 'A♦'],
                optimal: 'raise'
            },
            {
                id: 2,
                community: ['Q♣', 'J♣', 'T♣'],
                hand: ['9♣', '8♣'],
                optimal: 'call'
            },
            {
                id: 3,
                community: ['A♦', 'K♦', 'Q♦'],
                hand: ['J♦', 'T♦'],
                optimal: 'raise'
            }
        ];

        let currentScenario = 0;
        let score = 100;
        let totalPlays = 0;
        let correctPlays = 0;

        function showSection(section) {
            ['game-section', 'strategy-section', 'score-section'].forEach(s => {
                document.getElementById(s).style.display = 'none';
            });
            document.getElementById(`${section}-section`).style.display = 'block';
            
            if(section === 'score') updateScoreDisplay();
        }

        function updateScoreDisplay() {
            const percentage = totalPlays ? Math.round((correctPlays / totalPlays) * 100) : 100;
            document.getElementById('final-score').textContent = percentage;
            document.getElementById('score').textContent = percentage;
        }

        function loadScenario() {
            const scenario = scenarios[currentScenario];
            
            // Update community cards
            const commCards = document.getElementById('community-cards');
            commCards.innerHTML = scenario.community.map(card => 
                `<div class="card" data-suit="${card.slice(1)}">${card[0]}</div>`
            ).join('');

            // Update player hand
            const playerHand = document.getElementById('player-hand');
            playerHand.innerHTML = scenario.hand.map(card => 
                `<div class="card" data-suit="${card.slice(1)}">${card[0]}</div>`
            ).join('');

            // Clear feedback
            document.getElementById('feedback').style.display = 'none';
        }

        function handleAction(action) {
            const scenario = scenarios[currentScenario];
            totalPlays++;
            
            if(action === scenario.optimal) {
                correctPlays++;
                showFeedback('correct', `Correct! Optimal play is ${scenario.optimal.toUpperCase()} because...`);
            } else {
                score = Math.max(0, score - Math.floor(100/scenarios.length));
                showFeedback('incorrect', `Wrong! Optimal play is ${scenario.optimal.toUpperCase()}. ${getExplanation(scenario.id)}`);
            }
            
            updateScoreDisplay();
            currentScenario = (currentScenario + 1) % scenarios.length;
            setTimeout(loadScenario, 2000);
        }

        function showFeedback(type, message) {
            const feedback = document.getElementById('feedback');
            feedback.className = `feedback ${type}`;
            feedback.textContent = message;
            feedback.style.display = 'block';
        }

        function getExplanation(id) {
            const explanations = {
                1: "You have pocket Aces against three Kings - you're way ahead!",
                2: "You've flopped a straight flush draw - great pot odds to continue",
                3: "You've made the nut straight with Broadway"
            };
            return explanations[id] || "General optimal play strategy";
        }

        function resetScore() {
            score = 100;
            totalPlays = 0;
            correctPlays = 0;
            updateScoreDisplay();
        }

        // Initial load
        loadScenario();
    </script>
